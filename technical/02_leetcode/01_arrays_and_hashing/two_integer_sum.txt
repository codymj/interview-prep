class Solution
{
public:
    vector<int> twoSum(vector<int>& nums, int target)
    {
        unordered_map<int,int> comp;
        for (int i=0; i<nums.size(); ++i)
        {
            if (auto e = comp.find(nums[i]); e != comp.end())
            {
                return {e->second, i};
            }
            comp[target-nums[i]] = i;
        }
        return {};
    }
};

// 1. Two Sum
// https://leetcode.com/problems/two-sum

func twoSum(nums []int, target int) []int {
    // Map to hold element complements mapped to index.
    m := make(map[int]int)

    // Iterate over nums.
    for i1, e := range nums {
        comp := target - e

        // If the element is a complement of another element, found answer.
        if i2, found := m[e]; found {
            return []int{i1, i2}
        }

        // Otherwise, add complement to map.
        m[comp] = i1
    }

    return nil
}

class Solution
{
public:
    bool isAnagram(string const& s, string const& t)
    {
        if (s.size() != t.size()) return false;

        vector<int> v(26,0);
        for (int i=0; i<s.size(); ++i)
        {
            v[s[i]-'a']++;
            v[t[i]-'a']--;
        }
        for (int i=0; i<v.size(); ++i)
        {
            if (v[i] != 0) return false;
        }

        return true;
    }
};

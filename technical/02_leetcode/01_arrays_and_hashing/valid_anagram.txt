// 242. Valid Anagram
// https://leetcode.com/problems/valid-anagram

func isAnagram(s string, t string) bool {
    // Map to store letter counts.
    m := make(map[rune]int)

    // Iterate over s, incrementing counts.
    for _, v := range len(s) {
        m[v]++
    }

    // Iterate over t, decrementing count.
    // If count is already 0, no anagram.
    for _, v := range len(t) {
        if _, exists := m[v]; !exists || v == 0 {
            return false
        }
        m[v]--
    }

    // If at least one count is non-zero, no anagram.
    for _, v := range m {
        if v != 0 { return false }
    }

    return true
}

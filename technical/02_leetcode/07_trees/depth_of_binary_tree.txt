/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 * };
 */

class Solution
{
public:
    int maxDepth(TreeNode* root)
    {
        if (!root) return 0;

        int l_sub = maxDepth(root->left);
        int r_sub = maxDepth(root->right);

        return 1 + std::max(l_sub, r_sub);
    }
};

class Solution
{
public:
    bool isValid(string s)
    {
        stack<char> stk;
        for (auto ch : s)
        {
            if (ch == '(' || ch == '[' || ch == '{')
            {
                stk.push(s[i]);
                continue;
            }

            if (stk.size() == 0) return false;

            auto top = stk.top();
            if (ch == ')' && top != '(') return false;
            else if (ch == ']' && top != '[') return false;
            else if (ch == '}' && top != '{') return false;
            else stk.pop();
        }
    }
};
